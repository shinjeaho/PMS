import os
import pandas as pd
from collections import defaultdict

# 25개 대상 동 리스트
target_dongs = {
    ########## 25개동
}

# 경로 설정
base_dir = '작업해야하는 파일공간'
save_dir = '저장될 파일공간'
os.makedirs(save_dir, exist_ok=True)

# 처리할 종류
categories = ['신고공', '허가공']

# 각 동 폴더 순회
for dong in target_dongs:
    dong_path = os.path.join(base_dir, dong)
    if not os.path.isdir(dong_path):
        continue

    for category in categories:
        df_list = []

        # 해당 동 폴더 내의 신고공 또는 허가공 파일만 선택
        for file_name in os.listdir(dong_path):
            if not (file_name.startswith(category) and file_name.endswith('.xlsx')):
                continue

            file_path = os.path.join(dong_path, file_name)

            try:
                df = pd.read_excel(file_path)
                df_list.append(df)
                print(f"포함됨: {file_name} → {dong}/{category}")
            except Exception as e:
                print(f"읽기 실패: {file_name} - {e}")

        # 병합 및 중복 제거 후 저장
        if df_list:
            merged_df = pd.concat(df_list, ignore_index=True)
            deduped_df = merged_df.drop_duplicates()
            output_path = os.path.join(save_dir, f'{category}_{dong}.xlsx')
            deduped_df.to_excel(output_path, index=False)
            print(f"저장 완료: {output_path}")
